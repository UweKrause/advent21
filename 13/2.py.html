   27 def parse_input():
   28     dots = []
   29     folds = []
   30     with open(FILE) as lines:
   31         for line_raw in lines:
   32             line = line_raw.strip()
   33 
   34             if len(line) == 0:
   35                 continue
   36 
   37             if line.startswith("fold along "):
   38                 folds.append(line[11:])
   39                 continue
   40 
   41             x, y = line.split(",")
   42             dots.append((int(x), int(y)))
   43 
   44     return dots, folds
   45 
   46 
   47 def create_map(dots):
   48     max_x = -math.inf
   49     max_y = -math.inf
   50 
   51     for x, y in dots:
   52         max_x = max(max_x, x)
   53         max_y = max(max_y, y)
   54 
   55     shape = (max_x + 1, max_y + 1)
   56 
   57     a = np.zeros(shape=shape, dtype=int)
   58 
   59     for x, y in dots:
   60         a[x][y] = 1
   61 
   62     return a.transpose()
   63 
   64 
   65 def fold_it(folds, state):
   66     for fold in folds:
   67         axis, pos = fold.split("=")
   68 
   69         if axis == "y":
   70             state = fold_up(state, pos)
   71 
   72         if axis == "x":
   73             state = fold_left(state, pos)
   74 
   75     return state
   76 
   77 
   78 def fold_up(arr, pos):
   79     """folding is keeping the upper half of what you get if you flip the page (up) and stack it onto the original"""
   80 
   81     flip = np.flipud(arr)
   82     stack = arr + flip
   83 
   84     # return only the upper halve
   85     return np.split(stack, (0, int(pos)), axis=0)[1]
   86 
   87 
   88 def fold_left(arr, pos):
   89     """folding is keeping the left half of what you get if you flip the page (left) and stack it onto the original"""
   90 
   91     flip = np.fliplr(arr)
   92     stack = arr + flip
   93 
   94     # return only the left halve
   95     return np.split(stack, (0, int(pos)), axis=1)[1]
