1 from collections import defaultdict
    2 
    3 
    4 def main():
    5     """
    6     naive implementation, works, but takes ~20 Seconds on my machine
    7     """
    8     crabs = parse_input()
    9 
   10     pos_max = max(crabs)
   11 
   12     movement = {}
   13 
   14     for pos in range(pos_max + 1):
   15 
   16         pos_fuel = defaultdict(int)
   17 
   18         for crab in crabs:
   19             diff = max(crab, pos) - min(crab, pos)
   20             fuel = sum(range(diff + 1))
   21 
   22             # print(f"move from {crab} to {pos}: {fuel} fuel")
   23             pos_fuel[pos] += fuel
   24 
   25         # print(pos, sum(pos_fuel.values()))
   26 
   27         movement[pos] = sum(pos_fuel.values())
   28 
   29     print(min(movement.values()))  # 93397632
   30 
   31 
   32 def parse_input():
   33     with open("input") as lines:
   34         for line in lines:
   35             return [int(x) for x in line.strip().split(sep=",")]
   36 
   37 
   38 if __name__ == '__main__':
   39     main()
