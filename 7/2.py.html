1 import functools
    2 from collections import defaultdict
    3 
    4 FILE = "input"
    5 
    6 
    7 def main():
    8     """
    9     Naive implementation, works.
   10     Would take ~20 Seconds on my machine without memoization.
   11     Reduced to ~3 seconds with @functools.cache
   12     """
   13     crabs = parse_input()
   14 
   15     pos_max = max(crabs)
   16 
   17     movement = {}
   18 
   19     for pos in range(pos_max + 1):
   20 
   21         pos_fuel = defaultdict(int)
   22 
   23         for crab in crabs:
   24             diff = get_diff(crab, pos)
   25             fuel = calculate_distance(diff)
   26 
   27             # print(f"move from {crab} to {pos}: {fuel} fuel")
   28             pos_fuel[pos] += fuel
   29 
   30         # print(pos, sum(pos_fuel.values()))
   31 
   32         movement[pos] = sum(pos_fuel.values())
   33 
   34     return min(movement.values())
   35 
   36 
   37 @functools.cache
   38 def get_diff(crab, pos):
   39     return max(crab, pos) - min(crab, pos)
   40 
   41 
   42 @functools.cache
   43 def calculate_distance(diff):
   44     return sum(range(diff + 1))
   45 
   46 
   47 def parse_input():
   48     with open(FILE) as lines:
   49         for line in lines:
   50             return [int(x) for x in line.strip().split(sep=",")]
   51 
   52 
   53 if __name__ == '__main__':
   54     print(main())  # 93397632
