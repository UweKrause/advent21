   18         rows = 0
   19         fields = []
   20         for line in lines:
   21             rows += 1
   22             fields += line.strip()
   23 
   24         columns = len(line.strip())
   25 
   26         a = np.array(fields, dtype=int)
   27         a = a.reshape((rows, columns))
   28 
   29         return a

   32 def get_points_low(heightmap):
   33     rows, columns = heightmap.shape
   34     points_low = np.zeros(shape=(rows, columns), dtype=int)
   35     for r in range(rows):
   36         for c in range(columns):
   37             height = int(heightmap[r, c])
   38 
   39             height_left = int(heightmap[r, c - 1]) if (0 < c) else 9
   40             height_right = heightmap[r, c + 1] if (c < columns - 1) else 9
   41             height_up = heightmap[r - 1, c] if (0 < r) else 9
   42             height_down = heightmap[r + 1, c] if (r < rows - 1) else 9
   43 
   44             if height < height_left and height < height_right and height < height_up and height < height_down:
   45                 points_low[r, c] = 1
   46 
   47     return points_low
