   31 def parse_input():
   32     """parse input to numbers drawn and boards available"""
   33 
   34     with open("input") as lines:
   35 
   36         # first line contains all draws
   37         draws_all = [int(x) for x in lines.readline().split(sep=",")]
   38 
   39         # extract all the boards from remaining input
   40         boards = []
   41         while True:
   42 
   43             rows = []
   44 
   45             # read following lines from input
   46             # (1 empty line, 5 lines with content)
   47             for line in islice(lines, 1 + 5):
   48                 rows += [int(i) for i in line.split()]
   49 
   50             if rows:
   51                 assert len(rows) == 5 * 5
   52                 board = np.resize(np.array(rows), (5, 5))
   53                 boards.append(board)
   54             else:
   55                 # no input left, leave loop
   56                 break
   57 
   58     return draws_all, boards
   59 
   60 
   61 def is_board_winning(board, draws):
   62     """
   63     Checks if a board can win with regard to currently drawn numbers.
   64     A board wins if all of the numbers of one row or one column are drawn.
   65     """
   66 
   67     # The rows of a transposed board are the columns of the not transposed board
   68     return check_rows(board, draws) or check_rows(board.transpose(), draws)
   69 
   70 
   71 def check_rows(board, draws):
   72     for row in board:
   73         assert len(set(row)) == 5
   74         if set(row).issubset(draws):
   75             return True
   76     return False
   77 
   78 
   79 def sum_of_unmarked_fields(board, draws):
   80     """calculate the sum of all unmarked numbers of a winning board"""
   81     # todo: list comprehension
   82 
   83     unmarked = []
   84     for item in board.flat:
   85         if item not in draws:
   86             unmarked.append(item)
   87 
   88     return sum(unmarked)
